# Backend Development Patterns

## SQLAlchemy ORM Patterns

### Query Patterns

```python
from sqlalchemy.orm import Session

# Select specific columns (avoid SELECT *)
results = (
    db.query(Model.id, Model.score, Model.confidence)
    .filter(Model.scope_id == scope_id)
    .order_by(Model.score.asc())
    .limit(10)
    .all()
)

# Boolean/None comparisons (Ruff-enforced)
active = db.query(Model).filter(Model.enabled.is_(True)).all()
not_deleted = db.query(Model).filter(Model.deleted_at.is_(None)).all()

# Exists check (efficient)
from sqlalchemy import exists
has_results = db.query(
    exists().where(Model.scope_id == scope_id)
).scalar()
```

### N+1 Query Prevention

```python
from sqlalchemy.orm import joinedload, selectinload

# BAD: N+1 queries
items = db.query(Item).all()
for item in items:
    print(item.config.enabled)  # Each triggers a query

# GOOD: Eager load relationships
items = (
    db.query(Item)
    .options(joinedload(Item.config))
    .all()
)

# GOOD: Batch fetch for collections
items = (
    db.query(Item)
    .options(selectinload(Item.children))
    .all()
)
```

### Session Management

```python
from app.database import SessionLocal
from app.deps import get_db

# FastAPI dependency injection (preferred)
@router.get("/items/{item_id}")
def get_item(item_id: int, db: Session = Depends(get_db)):
    item = db.query(Item).filter_by(id=item_id).first()
    if not item:
        raise HTTPException(status_code=404)
    return item

# Manual session (tasks, scripts)
def process_in_task():
    db = SessionLocal()
    try:
        # Do work
        db.commit()
    except Exception:
        db.rollback()
        raise
    finally:
        db.close()
```

### Row Locking (with_for_update)

```python
# Prevent race conditions on concurrent updates
rec = (
    db.query(Model)
    .filter_by(id=rec_id)
    .with_for_update()
    .first()
)
if rec and rec.status == "PENDING":
    rec.status = "APPROVED"
    db.commit()
```

### JSONB Column Mutations

```python
from sqlalchemy.orm.attributes import flag_modified

# CRITICAL: SQLAlchemy does not detect in-place JSONB mutations
# ALWAYS call flag_modified() after mutating JSONB columns

# Appending to a list
rec.history.append({"status": "APPROVED"})
flag_modified(rec, "history")

# Setting a key in a dict
rec.metadata["reviewed_by"] = "admin"
flag_modified(rec, "metadata")

db.commit()
```

## Celery Task Patterns

### Basic Task with Retry

```python
from app.worker import celery_app

@celery_app.task(
    bind=True,
    max_retries=3,
    soft_time_limit=300,
    time_limit=600,
    acks_late=True,
)
def process_item(self, item_id: int) -> dict:
    try:
        db = SessionLocal()
        try:
            result = do_work(db, item_id)
            db.commit()
            return {"status": "ok", "item_id": item_id}
        finally:
            db.close()
    except TransientError as exc:
        raise self.retry(exc=exc, countdown=60 * (self.request.retries + 1))
```

### Task Configuration

```python
# ALWAYS use JSON serializer (never pickle)
celery_app.conf.task_serializer = "json"
celery_app.conf.result_serializer = "json"
celery_app.conf.accept_content = ["json"]
```

## Alembic Migration Patterns

```python
"""Add column to table."""

from alembic import op
import sqlalchemy as sa

def upgrade():
    op.add_column("items", sa.Column("confidence", sa.Float(), nullable=True))
    op.execute("UPDATE items SET confidence = 0.5 WHERE confidence IS NULL")
    op.alter_column("items", "confidence", nullable=False)

def downgrade():
    op.drop_column("items", "confidence")
```

### Migration Best Practices
- Always review autogenerated migrations before committing
- Test both `upgrade` and `downgrade` paths
- Commit migration with the corresponding model changes
- Never modify an existing migration after it has been applied

## Error Handling Patterns

### Centralized Error Handler

```python
from fastapi import HTTPException
from fastapi.responses import JSONResponse

@app.exception_handler(Exception)
async def global_exception_handler(request, exc):
    logger.error("Unhandled error: %s", exc, exc_info=True)
    return JSONResponse(
        status_code=500,
        content={"detail": "Internal server error"},
    )
```

### Retry with Exponential Backoff

```python
import time

def fetch_with_retry(fn, max_retries: int = 3):
    last_error = None
    for attempt in range(max_retries):
        try:
            return fn()
        except TransientError as e:
            last_error = e
            if attempt < max_retries - 1:
                delay = min(2 ** attempt * 1.0, 30.0)
                time.sleep(delay)
    raise last_error
```
